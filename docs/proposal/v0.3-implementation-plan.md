# 智引濠江 v0.3 Product Frontend Redesign Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use `superpowers:subagent-driven-development` (recommended) or `superpowers:executing-plans` to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Upgrade the v0.2 multi-route demo into a v0.3 role-based, visually credible, multi-end product frontend with explicit research, demo login, separate role shells, and preserved deterministic backend flow.

**Architecture:** Keep the existing FastAPI backend and React/Vite frontend. Phase 0 is a hard research checkpoint that selects the UI direction before page implementation; later phases add demo session, route guards, separate organizer/merchant/user shells, and role-specific pages while preserving the v0.2 API contracts.

**Tech Stack:** FastAPI, Pydantic, SQLite, pytest, React 19, Vite, TypeScript, Ant Design 6 currently installed, lucide-react, Vitest, Testing Library. Additional frontend UI libraries are not added until Phase 0 produces `docs/research/frontend-ui-decision.md` and the user approves it.

export type DemoRole = "organizer" | "merchant" | "tourist";

export interface DemoSession {
  role: DemoRole;
  actor_id: string;
  display_name: string;
  merchant_id?: string;
}

export const DEMO_SESSION_KEY = "zhiyin.demo.session";

export const DEMO_ACCOUNTS: Record<DemoRole, DemoSession> = {
  organizer: {
    role: "organizer",
    actor_id: "org-demo",
    display_name: "主办方演示账号"
  },
  merchant: {
    role: "merchant",
    actor_id: "merchant-m001",
    display_name: "福隆老饼家",
    merchant_id: "m001"
  },
  tourist: {
    role: "tourist",
    actor_id: "tourist-demo",
    display_name: "游客演示账号"
  }
};

export function getDemoSession(): DemoSession | null {
  const raw = localStorage.getItem(DEMO_SESSION_KEY);
  if (!raw) return null;
  try {
    const parsed = JSON.parse(raw) as DemoSession;
    if (parsed.role === "organizer" || parsed.role === "merchant" || parsed.role === "tourist") {
      return parsed;
    }
  } catch {
    localStorage.removeItem(DEMO_SESSION_KEY);
  }
  return null;
}

export function setDemoSession(session: DemoSession): void {
  localStorage.setItem(DEMO_SESSION_KEY, JSON.stringify(session));
}

export function clearDemoSession(): void {
  localStorage.removeItem(DEMO_SESSION_KEY);
}
```

import type { DemoRole, DemoSession } from "./session";

export function defaultPathForRole(role: DemoRole): string {
  if (role === "organizer") return "/organizer/dashboard";
  if (role === "merchant") return "/merchant/dashboard";
  return "/user/events/demo-night-tour";
}

export function requiredRoleForPath(pathname: string): DemoRole | "public" | null {
  if (pathname === "/" || pathname === "/login") return null;
  if (pathname.startsWith("/public/events/")) return "public";
  if (pathname.startsWith("/organizer") || pathname.startsWith("/review/")) return "organizer";
  if (pathname.startsWith("/merchant")) return "merchant";
  if (pathname.startsWith("/user")) return "tourist";
  return null;
}

export function canAccessPath(pathname: string, session: DemoSession | null): boolean {
  const required = requiredRoleForPath(pathname);
  if (required === null || required === "public") return true;
  return session?.role === required;
}
```

```tsx
import type { ReactNode } from "react";
import { getDemoSession } from "../auth/session";
import { canAccessPath } from "../auth/routeGuards";
import { ActivityWorkspacePage } from "../pages/ActivityWorkspacePage";
import { ExceptionCenterPage } from "../pages/ExceptionCenterPage";
import { LoginPage } from "../pages/login/LoginPage";
import { MerchantWorkbenchPage } from "../pages/MerchantWorkbenchPage";
import { OrganizerHomePage } from "../pages/OrganizerHomePage";
import { ReviewCenterPage } from "../pages/ReviewCenterPage";
import { TouristH5Page } from "../pages/TouristH5Page";

export type ShellKind = "organizer" | "merchant" | "user" | "public" | "login";

export interface ResolvedRoute {
  shell: ShellKind;
  activeKey: string;
  title: string;
  content: ReactNode;
}

export function resolveAppRoute(pathname: string): ResolvedRoute {
  const session = getDemoSession();
  const normalized = pathname === "/" ? "/login" : pathname;

  if (normalized === "/login") {
    return {
      shell: "login",
      activeKey: "login",
      title: "登录",
      content: <LoginPage />
    };
  }

  if (!canAccessPath(normalized, session)) {
    return {
      shell: "login",
      activeKey: "login",
      title: "登录",
      content: <LoginPage />
    };
  }

  if (normalized.startsWith("/public/events/")) {
    return {
      shell: "public",
      activeKey: "public",
      title: "游客入口",
      content: <TouristH5Page eventId={normalized.split("/")[3] || "demo-night-tour"} />
    };
  }

  if (normalized.startsWith("/user/events/")) {
    return {
      shell: "user",
      activeKey: "user-event",
      title: "活动路线",
      content: <TouristH5Page eventId={normalized.split("/")[3] || "demo-night-tour"} />
    };
  }

  if (normalized.startsWith("/merchant")) {
    const merchantId = session?.merchant_id || normalized.split("/")[2] || "m001";
    return {
      shell: "merchant",
      activeKey: "merchant-dashboard",
      title: "商户工作台",
      content: <MerchantWorkbenchPage merchantId={merchantId} />
    };
  }

  if (normalized.includes("/exceptions")) {
    return {
      shell: "organizer",
      activeKey: "exceptions",
      title: "异常中心",
      content: <ExceptionCenterPage eventId={normalized.split("/")[3] || "demo-night-tour"} />
    };
  }

  if (normalized.includes("/review") || normalized.startsWith("/review/")) {
    return {
      shell: "organizer",
      activeKey: "review",
      title: "复盘中心",
      content: <ReviewCenterPage eventId={normalized.split("/")[3] || normalized.split("/")[2] || "demo-night-tour"} />
    };
  }

  if (normalized.startsWith("/organizer/events/")) {
    return {
      shell: "organizer",
      activeKey: "workspace",
      title: "活动工作区",
      content: <ActivityWorkspacePage eventId={normalized.split("/")[3] || "demo-night-tour"} />
    };
  }

  return {
    shell: "organizer",
    activeKey: "dashboard",
    title: "活动运营",
    content: <OrganizerHomePage />
  };
}
```

Replace current route resolver in `apps/web/src/App.tsx` with:

```tsx
import { ConfigProvider } from "antd";
import { useEffect, useState } from "react";
import { appTheme } from "./theme";
import { resolveAppRoute } from "./routes/routeConfig";
import { OrganizerShell } from "./shells/OrganizerShell";
import { MerchantShell } from "./shells/MerchantShell";
import { UserShell } from "./shells/UserShell";

export default function App() {
  const [pathname, setPathname] = useState(window.location.pathname);
  const route = resolveAppRoute(pathname);

  useEffect(() => {
    const syncPath = () => setPathname(window.location.pathname);
    window.addEventListener("popstate", syncPath);
    return () => window.removeEventListener("popstate", syncPath);
  }, []);

  return (
    <ConfigProvider theme={appTheme}>
      {route.shell === "organizer" && (
        <OrganizerShell activeKey={route.activeKey} title={route.title}>
          {route.content}
        </OrganizerShell>
      )}
      {route.shell === "merchant" && (
        <MerchantShell activeKey={route.activeKey} title={route.title}>
          {route.content}
        </MerchantShell>
      )}
      {route.shell === "user" && (
        <UserShell activeKey={route.activeKey} title={route.title}>
          {route.content}
        </UserShell>
      )}
      {(route.shell === "public" || route.shell === "login") && route.content}
    </ConfigProvider>
  );
}
```

```tsx
import { Alert, Card, List, Space, Typography, message } from "antd";
import { useEffect, useState } from "react";
import { api } from "../../api";
import { RuntimeStatusControls } from "../../components/merchant/RuntimeStatusControls";
import type { Incident, MerchantRuntimeState, MerchantTask, MerchantWorkbench } from "../../types";

interface MerchantDashboardPageProps {
  merchantId: string;
}

export function MerchantDashboardPage({ merchantId }: MerchantDashboardPageProps) {
  const [workbench, setWorkbench] = useState<MerchantWorkbench>({ tasks: [] });
  const [incident, setIncident] = useState<Incident | null>(null);

  const load = async () => {
    const data = await api.getMerchantWorkbench(merchantId);
    setWorkbench({ ...data, tasks: data.tasks ?? [] });
  };

  useEffect(() => {
    load().catch(() => setWorkbench({ tasks: [] }));
  }, [merchantId]);

  const submitState = async (payload: Partial<MerchantRuntimeState>) => {
    const result = await api.updateRuntimeState(merchantId, payload);
    setWorkbench((current) => ({
      ...current,
      runtime_state: result.runtime_state || result
    }));
    setIncident(result.incident || null);
    if (result.incident) message.warning("状态已同步主办方异常中心");
  };

  return (
    <Space orientation="vertical" size={16} className="full-width">
      <Typography.Title level={2}>商户工作台</Typography.Title>
      <Alert type="info" showIcon title={workbench.merchant?.name || merchantId} description="这里是商户端，只处理今日任务和运行状态上报。" />
      <Card title="今日任务">
        <List
          dataSource={workbench.tasks}
          locale={{ emptyText: "暂无任务，主办方确认方案后会生成执行包。" }}
          renderItem={(task: MerchantTask) => (
            <List.Item>
              <List.Item.Meta title={`${task.role} / ${task.time_slot}`} description={task.visitor_task} />
            </List.Item>
          )}
        />
      </Card>
      <Card title="状态上报">
        <RuntimeStatusControls state={workbench.runtime_state} onSubmit={submitState} />
        {incident && <Alert className="mt-16" type="warning" showIcon title={`已触发 ${incident.type} 异常`} />}
      </Card>
    </Space>
  );
}
```
