# Zhiyin Haojiang v0.8 Agent Capability Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a backend-first, deterministic Agent capability layer that records specialist Agent runs, tool calls, controlled drafts, validation evidence, and human approval boundaries without making Qwen/QwenPaw or `DASHSCOPE_API_KEY` required.

**Architecture:** Extend the existing FastAPI/Pydantic/SQLite backend in place. Add first-class Agent contracts to the existing schema/store pattern, introduce a small deterministic Agent runtime around current tools and services, connect it to planning, incident recovery, and review flows, and keep frontend/runtime behavior backward compatible. Optional Qwen draft parsing is a later task after the deterministic Agent runtime is verified.

**Tech Stack:** FastAPI, Pydantic, SQLite record store, pytest, existing deterministic tools/services, optional DashScope-compatible Qwen adapter, existing React/Vite frontend unchanged for P0.

---

- No vector database or automatic web ingestion.

from collections.abc import Callable
from time import perf_counter
from typing import Any

from app.schemas import AgentToolCall


class ToolRecorder:
    def __init__(self, run_id: str):
        self.run_id = run_id
        self.calls: list[AgentToolCall] = []
        self._counter = 0

    def call(
        self,
        step_id: str,
        tool_name: str,
        input_payload: dict,
        fn: Callable[[], Any],
    ) -> Any:
        self._counter += 1
        started = perf_counter()
        call_id = f"tool_{self.run_id}_{self._counter:03d}"
        try:
            result = fn()
        except Exception as exc:
            elapsed_ms = int((perf_counter() - started) * 1000)
            self.calls.append(
                AgentToolCall(
                    tool_call_id=call_id,
                    run_id=self.run_id,
                    step_id=step_id,
                    tool_name=tool_name,
                    input_payload=input_payload,
                    output_payload={},
                    status="failed",
                    latency_ms=elapsed_ms,
                    error_summary=str(exc),
                )
            )
            raise
        elapsed_ms = int((perf_counter() - started) * 1000)
        self.calls.append(
            AgentToolCall(
                tool_call_id=call_id,
                run_id=self.run_id,
                step_id=step_id,
                tool_name=tool_name,
                input_payload=input_payload,
                output_payload={"result": result},
                status="success",
                latency_ms=elapsed_ms,
                error_summary=None,
            )
        )
        return result
```

from app.schemas import AgentEvaluation


PUBLIC_FORBIDDEN_TERMS = [
    "PlanVersion",
    "RecoveryProposal",
    "Incident",
    "AgentTrace",
    "backend",
    "schema",
    "fallback",
    "Qwen",
    "QwenPaw",
    "deterministic",
    "runtime state",
    "runtime_state",
]


def find_forbidden_public_terms(content: str) -> list[str]:
    lowered = content.lower()
    return [term for term in PUBLIC_FORBIDDEN_TERMS if term.lower() in lowered]


def evaluate_public_copy(
    run_id: str,
    content: str,
    human_approval_required: bool,
    fallback_used: bool,
) -> AgentEvaluation:
    forbidden = find_forbidden_public_terms(content)
    return AgentEvaluation(
        evaluation_id=f"eval_{run_id}_public_copy",
        run_id=run_id,
        schema_pass=bool(content.strip()),
        fallback_used=fallback_used,
        unsafe_mutation_attempted=False,
        human_approval_required=human_approval_required,
        forbidden_public_terms_present=bool(forbidden),
        public_copy_ready=bool(content.strip()) and not forbidden,
        notes=[f"forbidden_terms={','.join(forbidden)}"] if forbidden else ["public copy passed"],
    )
```

```python
import json
import os
from typing import Protocol

import httpx


class DraftBackend(Protocol):
    def generate_public_notice(self, event_id: str, incident_summary: str, recommended_action: str) -> dict:
        ...


class DeterministicDraftBackend:
    def generate_public_notice(self, event_id: str, incident_summary: str, recommended_action: str) -> dict:
        return {
            "content": "今晚路线已调整，请按最新路线前往下一站。",
            "locale": "zh-Hans",
            "structured_payload": {
                "event_id": event_id,
                "incident_summary": incident_summary,
                "recommended_action": recommended_action,
                "requires_organizer_approval": True,
            },
            "fallback_used": False,
            "response_status": "skipped",
        }


class QwenDraftBackend:
    def __init__(self):
        self.fallback = DeterministicDraftBackend()

    def generate_public_notice(self, event_id: str, incident_summary: str, recommended_action: str) -> dict:
        api_key = os.getenv("DASHSCOPE_API_KEY")
        model = os.getenv("QWEN_MODEL", "qwen-plus")
        if not api_key:
            result = self.fallback.generate_public_notice(event_id, incident_summary, recommended_action)
            return {**result, "fallback_used": True, "response_status": "provider_error"}
        try:
            response = httpx.post(
                "https://dashscope.aliyuncs.com/compatible-mode/v1/chat/completions",
                headers={"Authorization": f"Bearer {api_key}", "Content-Type": "application/json"},
                json={
                    "model": model,
                    "messages": [
                        {"role": "system", "content": "Return only JSON for a visitor-safe public notice draft."},
                        {
                            "role": "user",
                            "content": json.dumps(
                                {
                                    "event_id": event_id,
                                    "incident_summary": incident_summary,
                                    "recommended_action": recommended_action,
                                    "required_keys": ["content", "locale", "structured_payload"],
                                },
                                ensure_ascii=False,
                            ),
                        },
                    ],
                    "response_format": {"type": "json_object"},
                },
                timeout=30,
            )
            response.raise_for_status()
            content = response.json()["choices"][0]["message"]["content"]
            parsed = json.loads(content)
            if not isinstance(parsed.get("content"), str) or not parsed["content"].strip():
                raise ValueError("missing content")
            if parsed.get("locale") not in {"zh-Hans", "zh-Hant", "en", "mixed"}:
                raise ValueError("invalid locale")
            if not isinstance(parsed.get("structured_payload"), dict):
                raise ValueError("missing structured_payload")
            return {**parsed, "fallback_used": False, "response_status": "success"}
        except json.JSONDecodeError:
            result = self.fallback.generate_public_notice(event_id, incident_summary, recommended_action)
            return {**result, "fallback_used": True, "response_status": "invalid_json"}
        except Exception as exc:
            result = self.fallback.generate_public_notice(event_id, incident_summary, recommended_action)
            return {
                **result,
                "fallback_used": True,
                "response_status": "provider_error",
                "error_summary": str(exc),
            }


def choose_draft_backend() -> DraftBackend:
    backend = os.getenv("AGENT_DRAFT_BACKEND", "deterministic").lower()
    if backend == "qwen" and os.getenv("DASHSCOPE_API_KEY"):
        return QwenDraftBackend()
    return DeterministicDraftBackend()
```
