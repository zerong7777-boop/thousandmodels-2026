```python
from __future__ import annotations

import json
from typing import Any

from app.seed import seed_demo, seed_demo_accounts
from app.store import STORE


DEMO_ACCOUNTS = ["organizer.demo", "merchant.m001.demo", "tourist.demo"]


def reset_demo_state(event_id: str = "demo-night-tour") -> dict[str, Any]:
    STORE.clear_demo(event_id)
    if hasattr(STORE, "clear_auth_for_tests"):
        STORE.clear_auth_for_tests()
    if hasattr(STORE, "ensure_auth_schema"):
        STORE.ensure_auth_schema()
    seed_demo_accounts(STORE)
    event = seed_demo(STORE, event_id)
    return {
        "status": "reset",
        "event_id": event.event_id,
        "demo_accounts": DEMO_ACCOUNTS,
    }


def main() -> int:
    print(json.dumps(reset_demo_state(), ensure_ascii=False, indent=2))
    return 0


if __name__ == "__main__":
    raise SystemExit(main())
```

```typescript
const __filename = fileURLToPath(import.meta.url);
const __dirname = path.dirname(__filename);

const API_BASE = process.env.VITE_API_BASE_URL ?? "http://127.0.0.1:8000";
const EVENT_ID = "demo-night-tour";

async function apiPost<T>(
  request: APIRequestContext,
  pathName: string,
  data?: Record<string, unknown>
): Promise<T> {
  const response = await request.post(`${API_BASE}${pathName}`, {
    data,
    headers: { origin: "http://127.0.0.1:5179" }
  });
  expect(response.ok(), `${pathName}: ${response.status()} ${await response.text()}`).toBe(true);
  return response.json() as Promise<T>;
}

async function apiGet<T>(request: APIRequestContext, pathName: string): Promise<T> {
  const response = await request.get(`${API_BASE}${pathName}`);
  expect(response.ok(), `${pathName}: ${response.status()} ${await response.text()}`).toBe(true);
  return response.json() as Promise<T>;
}

async function setupDemo(request: APIRequestContext) {
  await apiPost(request, "/api/auth/login", {
```
